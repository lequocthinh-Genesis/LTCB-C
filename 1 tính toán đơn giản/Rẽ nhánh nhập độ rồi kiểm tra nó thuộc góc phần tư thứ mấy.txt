#include <stdio.h>
int main()
{
    int angle, k;
    scanf("%d", &angle);
    k = angle / 360;
    if(angle >= (360*k) && angle < (90 + 360*k))
        printf("1");
    else if (angle >= (90 + 360*k) && angle < (180 + 360*k))
        printf("2");
    else if (angle >= (180 + 360*k) && angle < (270 + 360*k))
        printf("3");
    else if (angle >= (270 + 360*k) && angle <= 360*(k + 1))
        printf("4");
    return 0;
}
