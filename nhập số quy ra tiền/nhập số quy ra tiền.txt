/** Chap vao 1 so co co 1,2 hay nhieu chu so va <100.000 va doc so vua nhap vao **/

int main()
{
    int n;
    scanf("%d",&n);

    //Tach cac so lan luot o hang tram nghin, chuc nhin, hang nghin, hang tram, hang chuc, hang don vi
    int ti=(n/1000000000);
	int ttr=(n/100000000)%10;
	int ctr=(n/10000000)%10;
	int tr=(n/1000000)%10;
	int tng=(n/100000)%10;
    int cng=(n/10000)%10;
    int ng=(n/1000)%10;
    int t=(n/100)%10;
    int c=(n/10)%10;
    int dv=n%10;

    //Doc cac so vua tach o tren
    printf("\nDoc: ");
		if(n== 1000000000){
			printf("Mot Ty dong");
		}
		else if(n== 100000000){
			printf("Mot Tram Trieu dong");
		}
		else if(n==10000000){
			printf("Muoi Trieu dong");
		}
		else if(n== 1000000){
			printf("Mot Trieu dong");
		}
		else if(n==100000){
			printf("Mot Tram Nghin dong");
		}
		else if(n== 10000){
			printf("Muoi Nghin dong");
		}
		else if(n==1000){
			printf("Mot Nghin dong");
		}
		else if(n== 100){
			printf("Mot tram dong");
		}
		else{
    		ty(ti);			printf(" ");
    		tramtrieu(ttr); printf(" ");
    		chuctrieu(ctr); printf(" ");
			trieu(tr);      printf(" ");
    		tramnghin(tng); printf(" ");
    		chucnghin(cng); printf(" ");
    		nghin(ng);      printf(" ");
    		tram(t);        printf(" ");
    		chuc(c);        printf(" ");
    		donvi(dv);      printf(" ");
    	}

    return 0;
}
